Implementation Specifications

  • Grid Dimensions: 80 rows × 52 columns = 4,160 total cells
  • Cell Size: 8px × 8px with 2px gap
  • Filled Cells: Mori Gold (#D4AF37) — represents lived weeks
  • Current Week: Ember Orange (#FF6B35) with pulse animation
  • Empty Cells: Border Light (#E8E8E8) — represents remaining weeks
  • Interaction: Tap cell to see date range (Week X, Year Y)
  • Responsiveness: Horizontal scroll on mobile, full view on tablet+
  • Animation: Hover scale (1.5x), current week pulse (2s infinite)
  • Accessibility: Each cell has ARIA label with week number and date range 
// SwiftUI Component: LifeGridView.swift
struct LifeGridView: View {
    let birthDate: Date
    let lifeExpectancy: Int // default 80
    
    var body: some View {
        ScrollView([.horizontal]) {
            LazyVGrid(columns: Array(repeating: GridItem(.fixed(8), spacing: 2), count: 52), spacing: 2) {
                ForEach(0..<(lifeExpectancy * 52), id: \.self) { weekIndex in
                    LifeCell(
                        weekIndex: weekIndex,
                        isFilled: isWeekLived(weekIndex),
                        isCurrent: isCurrentWeek(weekIndex)
                    )
                }
            }
        }
    }
}
